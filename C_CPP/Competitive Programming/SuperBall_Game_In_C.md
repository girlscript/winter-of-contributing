                                                delay(5);

                                                outtext("SuperBall");      // Text to be displayed on the screen
                                                delay(5);

                                    outtextxy(220,400,"DO YOU WANT TO PLAY THE GAME");

                                    ch=getch();

                                    goto aage;
                                    if(ch==27)
                                    return;
                                    if(ch==110||ch==78)
                                    return;
                                    else
                                    return;
                                    aage:
                                    i=250;
                                    x=300;
                                    y=415;
                                    flag=0;
                                    left=0;

                                    while(1)
                                    {

                                    outtextxy(130 ,460,"Developer: Kavya Jain"); 
                                    if(x>=614||x<=25)
                                    flag=!flag;
                                    if(flag)
                                    x=x-1;
                                    else
                                    x=x+1;
                                    if((y>=416&&x<=i+100&&x>=i)||y<=35)
                                    left=!left;
                                    if(left)
                                    y=y+1;
                                    else
                                    y=y-1;
                                    if(y==415&&x<=i+100&&x>=i) SCORE[0]++;
                                    moveto(600,5);
                                    sprintf
                                    (str,"%d", SCORE[0]);
